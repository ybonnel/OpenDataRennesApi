14  package fr.ybo.opendata.rennes.util;

16  /**
17   * Classe utilitaire pour les formatage d'une chaine.
18   */
19  public final class Formatteur {
20  
21      /**
22       * Constructeur privé.
23       */
24      private Formatteur() {
25      }
26  
27      /**
28       * Permet de formatter une chaîne de caractère :
29       * <ul>
30       *     <li>Pour chaque mot : une majuscule au début et le reste un minuscule.</li>
31       *     <li>Tout les double espaces sont remplacer par un seul espace.</li>
32       *     <li>S'il y a des espaces au début de la chaîne, ils sont supprimés.</li>
33       * </ul>
34       * @param chaine chaîne à formatter.
35       * @return la chaîne formattée.
36       */
37      public static String formatterChaine(String chaine) {
38          StringBuilder nomLongFormateBuilder = new StringBuilder();
39          for (String champ : chaine.replaceAll("/", "-").split(" ")) {
40              for (String champ2 : champ.split("\\(")) {
41                  if (champ2.length() > 0) {
42                      nomLongFormateBuilder.append(champ2.substring(0, 1).toUpperCase());
43                      nomLongFormateBuilder.append(champ2.substring(1, champ2.length()).toLowerCase());
44                  }
45                  nomLongFormateBuilder.append('(');
46              }
47              // on enleve le dernier tiret.
48              nomLongFormateBuilder.deleteCharAt(nomLongFormateBuilder.length() - 1);
49              nomLongFormateBuilder.append(' ');
50          }
51          // on enleve le dernier espace.
52          nomLongFormateBuilder.deleteCharAt(nomLongFormateBuilder.length() - 1);
53          String nomLongFormate = nomLongFormateBuilder.toString().replaceAll("\\|", "");
54          while (nomLongFormate.contains("  ")) {
55              nomLongFormate = nomLongFormate.replaceAll("  ", " ");
56          }
57          while (nomLongFormate.length() > 0 && nomLongFormate.charAt(0) == ' ') {
58              nomLongFormate = nomLongFormate.substring(1);
59          }
60          return nomLongFormate;
61      }
62  }
